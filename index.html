    默认调用

    $("waterfall1").waterfall();

/* 默认值如下:
{
  item: "div", // String 子元素的选择器,默认是 div
  colMinWidth: 320, // Nunber 每一列的宽度, 默认是 320
  colGapWidth: 20, // Number 列之间的宽度, 默认是 20
  autoResize: false // Bool 是否自动响应容器宽度, 默认为 false
}*/
